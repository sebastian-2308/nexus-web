def calcular_rango(xp):

        return "🌑 Novato"
    elif xp <= 499:
        return "🌙 Astrónomo"
    elif xp <= 999:
        return "🚀 Explorador"
    else:
        return "👑 Leyenda Cósmica"

# ¡Prueba con tu XP!
